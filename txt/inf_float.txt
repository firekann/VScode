void ltrim(std::string& s)
{
    s.erase(s.begin(), find_if(s.begin(), s.end(), [](int ch) { return ((ch == '0' || ch == ' ') ? false : true); }));
}
void rtrim(std::string& s)
{
    s.erase(find_if(s.rbegin(), s.rend(), [](int ch) { return ((ch == '0' || ch == ' ') ? false : true); }).base(), s.end());
}
void trim(std::string& s)
{
    rtrim(s);
    ltrim(s);
}

void FFT(std::vector<std::complex<double>>& f, bool inv = false)
{
    int size = f.size();
    for (int i = 1, j = 0; i < size; i++) {
        int tmp = size / 2;
        while (!((j ^= tmp) & tmp)) {
            tmp /= 2;
        }
        if (i < j) {
            std::swap(f[i], f[j]);
        }
    }
    for (int i = 1; i < size; i *= 2) {
        double tmp = (inv ? acos(-1) / i : -acos(-1) / i);
        std::complex<double> w(cos(tmp), sin(tmp));
        for (int j = 0; j < size; j += i * 2) {
            std::complex<double> wp(1, 0);
            for (int k = 0; k < i; k++) {
                std::complex<double> x = f[j + k], y = f[i + j + k] * wp;
                f[j + k] = x + y;
                f[i + j + k] = x - y;
                wp *= w;
            }
        }
    }
    if (inv) {
        for (int i = 0; i < size; i++) {
            f[i] /= size;
        }
    }
}

bool operator>(const inf_float& a, const inf_float& b)
{
    if (a == b) {
        return false;
    }
    bool abs = false;
    if ((int)a.length - (int)a.precision_position > (int)b.length - (int)b.precision_position) {
        abs = true;
    } else if ((int)a.length - (int)a.precision_position == (int)b.length - (int)b.precision_position) {
        int i = (int)a.length;
        while (i--) {
            int tmp = a.digits[i] - b.digits[i];
            if (tmp == 0) {
                continue;
            } else if (tmp > 0) {
                abs = true;
                break;
            } else {
                abs = false;
                break;
            }
        }
    }
    return a.thesign ? abs : !abs;
}

inf_float operator+(const inf_float& a, const inf_float& b)
{
    if (a.thesign != b.thesign && a.digits == b.digits && a.precision_position == b.precision_position) {
        return 0;
    }
    if (a.thesign == true && b.thesign == true) {
        unsigned int max_precision_position = max(a.precision_position, b.precision_position);
        std::string sa = a.digits;
        std::string sb = b.digits;
        if (a.precision_position < b.precision_position) {
            for (int i = 0; i < b.precision_position - a.precision_position; i++) {
                sa = '0' + sa;
            }
        } else {
            for (int i = 0; i < a.precision_position - b.precision_position; i++) {
                sb = '0' + sb;
            }
        }
        int a_len = (int)sa.length();
        int b_len = (int)sb.length();
        while ((int)sb.length() < a_len) {
            sb += '0';
        }
        while ((int)sa.length() < b_len) {
            sa += '0';
        }
        std::string result = "";
        int carry = 0, num;
        for (int i = 0; i < (int)sa.length(); i++) {
            num = (sa[i] - '0' + sb[i] - '0' + carry);
            result += ((num % 10) + '0');
            carry = num / 10;
        }
        if (carry) {
            result += (carry + '0');
        }
        inf_float res;
        int len = (int)result.length();
        if (max_precision_position == 0) {
            res.digits = result;
            res.length = len;
        } else if (len > max_precision_position) {
            int pos = -1;
            for (int i = 0; i < len; i++) {
                if (i == max_precision_position) {
                    break;
                }
                if (result[i] != '0') {
                    pos = i;
                    break;
                }
            }
            if (pos == -1) {
                pos = max_precision_position;
            }
            result = result.substr(pos);
            res.length = (unsigned int)result.length();
            res.precision_position = max_precision_position - (len - (int)res.length);
            res.digits = result;
        } else {
            ltrim(result);
            res.length = (unsigned int)result.length();
            res.precision_position = max_precision_position - (len - (int)res.length);
            res.digits = result;
        }
        return res;
    } else if (a.thesign == false && b.thesign == false) {
        return -(abs(a) + abs(b));
    } else {
        inf_float c(b);
        c.thesign = a.thesign;
        return a - c;
    }
}
inf_float operator-(const inf_float& a, const inf_float& b)
{
    if (a == b) {
        return 0;
    }
    if (a.thesign == true && b.thesign == false) {
        return a + abs(b);
    } else if (a.thesign == false && b.thesign == true) {
        return -(abs(a) + b);
    } else if (a.thesign == false && b.thesign == false) {
        return -(abs(a) - abs(b));
    } else if (a.thesign == true && b.thesign == true) {
        unsigned int max_precision_position = max(a.precision_position, b.precision_position);
        bool flag = false;
        inf_float aa(a);
        inf_float bb(b);
        if (aa < bb) {
            aa = b;
            bb = a;
            flag = true;
        }
        std::string sa = aa.digits;
        std::string sb = bb.digits;
        if (aa.precision_position < bb.precision_position) {
            for (int i = 0; i < bb.precision_position - aa.precision_position; i++) {
                sa = '0' + sa;
            }
        } else {
            for (int i = 0; i < aa.precision_position - bb.precision_position; i++) {
                sb = '0' + sb;
            }
        }
        int abs_len = abs((int)((int)sa.length() - (int)sb.length()));
        for (int i = 0; i < abs_len; i++) {
            sb += '0';
        }
        for (auto& i : sb) {
            i = '9' - i + '0';
        }
        sb[0]++;
        std::reverse(sa.begin(), sa.end());
        std::reverse(sb.begin(), sb.end());
        inf_float result(inf_float(sa) + inf_float(sb));
        result.digits.erase(result.digits.end() - 1);
        rtrim(result.digits);
        std::string tmp = result.digits;
        rtrim(tmp);
        inf_float res;
        if (flag) {
            res.thesign = false;
        }
        int len = (int)tmp.length();
        if (max_precision_position == 0) {
            res.digits = tmp;
            res.length = len;
        } else if (len > max_precision_position) {
            int pos = -1;
            for (int i = 0; i < len; i++) {
                if (i == max_precision_position) {
                    break;
                }
                if (tmp[i] != '0') {
                    pos = i;
                    break;
                }
            }
            if (pos == -1) {
                pos = max_precision_position;
            }
            tmp = tmp.substr(pos);
            res.length = (unsigned int)tmp.length();
            res.precision_position = max_precision_position - (len - (int)res.length);
            res.digits = tmp;
        } else {
            ltrim(tmp);
            res.length = (unsigned int)tmp.length();
            res.precision_position = max_precision_position - (len - (int)res.length);
            res.digits = tmp;
        }
        return res;
    }
}
inf_float operator*(const inf_float& a, const inf_float& b)
{
    if (a == 0 || b == 0) {
        return 0;
    }
    if (a == 1 || a == -1) {
        return a.thesign ? b : -b;
    }
    if (b == 1 || b == -1) {
        return b.thesign ? a : -a;
    }
    if (a.thesign != b.thesign) {
        return -(abs(a) * abs(b));
    } else if (a.thesign == false && b.thesign == false) {
        return abs(a) * abs(b);
    } else {
        std::vector<std::complex<double>> A, B, C;
        for (auto i : a.digits) {
            A.push_back(std::complex<double>(i - '0', 0));
        }
        for (auto i : b.digits) {
            B.push_back(std::complex<double>(i - '0', 0));
        }
        int num = 1;
        int Asize = A.size();
        int Bsize = B.size();
        while (num < Asize + 1 || num < Bsize + 1) {
            num *= 2;
        }
        num *= 2;
        A.resize(num);
        B.resize(num);
        C.resize(num);
        FFT(A);
        FFT(B);
        for (int i = 0; i < num; i++) {
            C[i] = A[i] * B[i];
        }
        FFT(C, true);
        int carry = 0;
        std::vector<int> r;
        int Csize = C.size();
        for (int i = 0; i < Csize; i++) {
            int tmp = (int)round(C[i].real()) + carry;
            r.push_back(tmp % 10);
            carry = tmp / 10;
        }
        while (carry > 0) {
            r.push_back(carry % 10);
            carry /= 10;
        }
        int rsize = r.size();
        int i;
        for (i = rsize - 1; i >= 0 && r[i] == 0; i--) { }
        std::string result = "";
        if (i < 0) {
            return 0;
        } else {
            for (; i >= 0; i--) {
                result += (r[i] + '0');
            }
        }
        unsigned int precision_sum = a.precision_position + b.precision_position;
        int len = (int)result.length();
        if (precision_sum == 0) {
            return result;
        } else if (len > precision_sum) {
            int pos = -1;
            for (int i = len - 1; i >= 0; i--) {
                if (i == precision_sum) {
                    break;
                }
                if (result[i] != '0') {
                    pos = i;
                    break;
                }
            }
            if (pos == -1) {
                pos = precision_sum;
            }
            result = result.substr(0, pos);
            inf_float res(result);
            res.precision_position = precision_sum - (len - (int)res.length);
            return res;
        } else {
            rtrim(result);
            inf_float res(result);
            res.precision_position = precision_sum - (len - (int)res.length);
            return res;
        }
    }
}

std::ostream& operator<<(std::ostream& out, const inf_float& a)
{
    if (a.thesign == false) {
        out << '-';
    }
    if (a.output_precision_length == 0) {
        if (a.length > a.precision_position) {
            for (int i = (int)a.length - 1; i >= (int)a.length - ((int)a.length - (int)a.precision_position); i--) {
                out << a.digits[i];
            }
        } else {
            out << "0";
        }
    } else {
        bool is_precision = false;
        int cnt = 0;
        if (a.length <= a.precision_position) {
            is_precision = true;
            out << "0.";
            for (int i = 0; i < (int)a.precision_position - (int)a.length; i++) {
                out << "0";
                cnt++;
            }
        }
        for (int i = (int)a.length - 1; i >= 0; i--) {
            if (cnt == a.output_precision_length) {
                break;
            }
            if (is_precision) {
                cnt++;
            }
            out << a.digits[i];
            if (i == a.precision_position) {
                is_precision = true;
                out << ".";
                if (i == 0) {
                    out << "0";
                }
            }
        }
        if (a.output_precision_length != -1 && a.precision_position < a.output_precision_length) {
            for (int i = 0; i < a.output_precision_length - (a.precision_position == 0 ? 1 : a.precision_position); i++) {
                out << "0";
            }
        }
    }
    return out;
}
